\begin{frame}{The Computational Bottleneck}
    \framesubtitle{Why Classical Computers Struggle with Crop Rotation}
    
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \textbf{Problem Structure:}
            \begin{itemize}
                \item \textbf{Quadratic objective} from rotation synergies
                \item \textbf{Frustrated constraints}: 70--88\% of crop pairs have \textit{negative} interactions
                \item Creates rugged energy landscape with many local minima
            \end{itemize}
            
            % \vspace{0.3cm}
            \textbf{Classical Solver Barriers:}
            \begin{enumerate}\footnotesize
                \item Quadratic terms break LP relaxation
                \item Poor bounds $\Rightarrow$ no effective pruning
                \item Branch-and-bound explores exponentially
            \end{enumerate}
            
            % \vspace{0.3cm}
            \begin{warnbox}
                \footnotesize\textbf{Result:} Gurobi times out on \textbf{85\%} of rotation scenarios
            \end{warnbox}
        \end{column}
        
        \begin{column}{0.43\textwidth}
            \textbf{Quantum Advantage Mechanism:}
            
            \vspace{0.2cm}
            \begin{tikzpicture}[scale=0.9]
                % Energy landscape visualization
                \draw[thick,primaryBlue] plot[smooth,tension=0.7] coordinates {
                    (0,2) (0.5,1.5) (1,2.2) (1.5,1) (2,1.8) (2.5,0.5) (3,1.2) (3.5,1.5) (4,2)
